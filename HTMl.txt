    // 🔐 LOGIN
    const usuarioCorreto = "Matheus";
    const senhaCorreta = "inf12";

    function logar() {
        let user = document.getElementById("usuario").value.trim();
        let pass = document.getElementById("senha").value.trim();

        if (user === usuarioCorreto && pass === senhaCorreta) {
            document.getElementById("login").classList.remove("ativa");
            document.getElementById("loja").classList.add("ativa");
        } else {
            document.getElementById("erroLogin").innerHTML = "❌ Login inválido!";
        }
    }

    // 🛒 SISTEMA
    const nomeCliente = "Matheus";
    const endereco = "Rua 1, 123, Ribeirão das Neves ";
    let saldo = 100000000;

    function comprar(escolha) {

        let preco;
        let perfume;
        let quantidade;

        if (escolha === 1) {
            perfume = "Sakura";
            preco = 80;
            quantidade = document.getElementById("qtd1").value;

        } else if (escolha === 2) {
            perfume = "Tokyo Night";
            preco = 120;
            quantidade = document.getElementById("qtd2").value;

        } else {
            perfume = "Imperial Gold";
            preco = 150;
            quantidade = document.getElementById("qtd3").value;
        }

        // valida quantidade
        if (!quantidade || quantidade <= 0) {
            document.getElementById("resultado").innerHTML = "⚠️ Coloque uma quantidade válida!";
            return;
        }

        // --- LÓGICA DO DESCONTO CONDICIONAL ---
        let totalOriginal = preco * quantidade;
        let porcentagemDesconto = 0; 

        // Se atingir ou passar de 200 reais, a porcentagem muda para 10
        if (totalOriginal >= 200) {
            porcentagemDesconto = 10;
        }

        let valorDoDesconto = (totalOriginal * porcentagemDesconto) / 100;
        let totalComDesconto = totalOriginal - valorDoDesconto;

        // --- MENSAGEM DINÂMICA COM OS VALORES ---
        let mensagem = "Cliente: " + nomeCliente + "<br>";
        mensagem += "Perfume: " + perfume + "<br>";
        mensagem += "Quantidade: " + quantidade + "<br>";
        mensagem += "Preço bruto: R$ " + totalOriginal + "<br>";
        
        // Só mostra a linha de desconto na tela se ele for maior que zero
        if (porcentagemDesconto > 0) {
            mensagem += "🎁 Desconto aplicado (10%): -R$ " + valorDoDesconto + "<br>";
        } else {
            mensagem += "ℹ️ Desconto não aplicado (compras abaixo de R$ 200)<br>";
        }
        
        mensagem += "Total a pagar: R$ " + totalComDesconto + "<br>";

        // Validação usando o valor com desconto
        if (saldo >= totalComDesconto) {
            saldo -= totalComDesconto;
            mensagem += "✅ Compra realizada!<br>";
            mensagem += "Entrega em: " + endereco + "<br>";
            mensagem += "💰 Saldo restante: R$ " + saldo;
        } else {
            mensagem += "❌ Saldo insuficiente!";
        }

        document.getElementById("resultado").innerHTML = message = mensagem;
    }
